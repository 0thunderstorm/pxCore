# USAGE !
#
# rasterDFB.sh -?

USAGE:  rasterDFB <args> 

        -w       Show "work:"  per frame timings
        -r       Use rotation Static of 15 Degrees
        -a       Use rotation Animation
        -b       Skip Alpha Blending
        -c       Skip clearscreen
        -d       Skip *ALL* Drawing
        -m       Skip *ALL* Matrix Math
        -x       Drag race... Don't SLEEP !
        -xl NN   Drag race... NN times in one process
        -vm      Use VIDEO memory for images
        -csv     Output format in CSV
        -s NN    Use NN for square dimension
        -i NN    Exit after 'NN' Avg. Windows data points 
        -t NN    Exit after 'NN ms' (Milliseconds) 
        -n NN    Use NN squares in checherboard
        -y NN    Use NN for OversampleY setting
        -f NN    Use NN for FPS rate
        -w NN    Use NN for FPS Avg. Window size

        -fg CLR  Use Foreground color   (WHITE,BLACK,BLUE,RED,GREEN,GRAY)
        -bg CLR  Use Background color   (WHITE,BLACK,BLUE,RED,GREEN,GRAY)

        Use "mandel"  for Mandelbrot Set texture fill.
        Use "solid"   for Solid Color    texture fill.
